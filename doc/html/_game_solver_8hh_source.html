70 WinningDomain solve_reachability_game(const TransitionFunction& trans_function, F1& target, F2&& avoid, val_functionction_type& value) {
71  /* size of state alphabet */
72  abs_type N=trans_function.m_no_states;
73  /* size of input alphabet */
74  abs_type M=trans_function.m_no_inputs;
75 
76  abs_type max = std::numeric_limits<abs_type>::max();
77  if(M > max-1) {
78  throw std::runtime_error("scots::solve_reachability_game: Number of inputs exceeds maximum supported value");
79  }
80  /* win_domain[i] = j
81  * contains the input j associated with state i
82  * j = max if the target is not reachable from i
83  *
84  * initialize all states to loosing (win_domain[i]=max) */
85  std::vector<abs_type> win_domain(N,max);
86  /* initialize value */
87  value.resize(N,std::numeric_limits<double>::infinity());
88  /* keep track of the number of processed post */
89  abs_type* K = new abs_type[N*M];
90  /* keep track of the values (corresponds to M in Alg.2)*/
91  double* edge_val = new double[N*M];
92 
93  /* init fifo */
94  std::queue<abs_type> fifo;
95  for(abs_type i=0; i<N; i++) {
96  if(target(i) && !avoid(i)) {
97  win_domain[i]=0;
98  /* states in the target set have value zero */
99  value[i]=0;
100  /* states in the target are added to the fifo */
101  fifo.push(i);
102  }
103  for(abs_type j=0; j<M; j++) {
104  edge_val[i*M+j]=0;
105  K[i*M+j]=trans_function.m_no_post[i*M+j];
106  }
107  }
108 
109  /* main loop */
110  while(!fifo.empty()) {
111  /* get state to be processed */
112  abs_type q=fifo.front();
113  fifo.pop();
114  /* loop over each input */
115  for(abs_type j=0; j<M; j++) {
116  /* loop over pre's associated with this input */
117  for(abs_ptr_type v=0; v<trans_function.m_no_pre[q*M+j]; v++) {
118  abs_type i=trans_function.m_pre[trans_function.m_pre_ptr[q*M+j]+v];
119  if(avoid(i))
120  continue;
121  /* (i,j,q) is a transition */
122  /* update the number of processed posts */
123  K[i*M+j]--;
124  /* update the max value of processed posts */
125  edge_val[i*M+j]=(edge_val[i*M+j]>=1+value[q] ? edge_val[i*M+j] : 1+value[q]);
126  /* check if for node i and input j all posts are processed */
127  if(!K[i*M+j] && value[i]>edge_val[i*M+j]) {
128  fifo.push(i);
129  value[i]=edge_val[i*M+j];
130  win_domain[i]=j;
131  }
132  } /* end loop over all pres of state i under input j */
133  } /* end loop over all input j */
134  } /* fifo is empty */
135  delete[] K;
136  delete[] edge_val;
137 
138  return WinningDomain(N,M,std::move(win_domain));
139 }

150 template<class F>
151 WinningDomain solve_invariance_game(const TransitionFunction& trans_function, F& safe) {
152  /* size of state alphabet */
153  abs_type N=trans_function.m_no_states;
154  /* size of input alphabet */
155  abs_type M=trans_function.m_no_inputs;
156  /* helper */
157  abs_type max = std::numeric_limits<abs_type>::max();
158 
159  /* valid_inputs
160  * boolean array of size N*M
161  * valid_inputs[i*M+j]=true iff input j
162  * is a valid input at state i */
163  std::vector<bool> valid_inputs(N*M,false);
164  /* no_input: keep track of the number of valid inputs.
165  * If no_val_in[i]==0, then state i is not winning and
166  * no_val_in[i] is set to max (see also WinningDomain) */
167  std::vector<abs_type> no_val_in(N,0);
168  /* keep track if an unsafe state was already added to the fifo */
169  std::vector<bool> added(N,false);
170 
171  /* initialization */
172  std::queue<abs_type> fifo;
173  for(abs_type i=0; i<N; i++) {
174  if(safe(i)) {
175  for(abs_type j=0; j<M; j++) {
176  if(trans_function.m_no_post[i*M+j]) {
177  valid_inputs[i*M+j]=true;
178  no_val_in[i]++;
179  }
180  }
181  }
182  if(!no_val_in[i]) {
183  fifo.push(i);
184  added[i]=true;
185  /* mark no_val_in[i]=max to indicate that state i ist not winning */
186  no_val_in[i]=max;
187  }
188  }
189 
190  while(!fifo.empty()) {
191  abs_type k=fifo.front();
192  fifo.pop();
193  /* loop over all inputs */
194  for(abs_type j=0; j<M; j++) {
195  /* loop over all pre states of (k,j) */
196  for(abs_ptr_type p=0; p<trans_function.m_no_pre[k*M+j]; p++) {
197  /* (i,j,k) is a transition */
198  abs_type i=trans_function.m_pre[trans_function.m_pre_ptr[k*M+j]+p];
199  /* check if input j at state i is considered safe */
200  if(valid_inputs[i*M+j]) {
201  /* set source states i with label j as unsafe pair */
202  valid_inputs[i*M+j]=false;
203  no_val_in[i]--;
204  }
205  /* add to unsafe set if state i has no more valid inputs */
206  if(!no_val_in[i] && !added[i]) {
207  fifo.push(i);
208  added[i]=true;
209  /* mark no_val_in[i]=max to indicate that state i ist not winning */
210  no_val_in[i]=max;
211  }
212  }
213  }
214  }
215  return WinningDomain(N,M,std::move(no_val_in),std::move(valid_inputs));
216 }
