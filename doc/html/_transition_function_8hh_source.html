143  std::vector<abs_type> pre{};
144  if(m_pre!=nullptr) {
145  if(!m_no_pre[k*m_no_inputs+j]) {

160  std::vector<abs_type> get_post(const abs_type& i, const abs_type& j) const {
161  std::vector<abs_type> post{};
162  if(m_pre!=nullptr) {
163  for(abs_type k=0; k<m_no_states; k++) {
164  if(m_no_pre[k*m_no_inputs+j]) {

166  abs_type pos=m_pre_ptr[k*m_no_inputs+j]+no;
167  if(m_pre[pos]==i) {
