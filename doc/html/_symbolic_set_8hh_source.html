184  std::vector<double> bdd_to_grid_points(const Cudd& manager, BDD bdd) {
185  if((!get_no_bdd_vars()) || bdd==manager.bddZero())
186  return {};
187  /* disable reordering (if enabled) */
188  Cudd_ReorderingType *method=nullptr;
189  if(manager.ReorderingStatus(method))
190  manager.AutodynDisable();
191  /* get variable ids */
192  auto var_id = get_bdd_var_ids();
193  /* find the variables in the support of the BDD but outside the SymbolicSet */
194  auto support_id = bdd.SupportIndices();
195  std::vector<BDD> out{};
196  for(const auto& id : support_id) {
197  if(std::find(std::begin(var_id), std::end(var_id), id)==std::end(var_id))
198  out.emplace_back(manager.bddVar(id));
199  }
200  /* remove those variables from the bdd */
201  if(out.size())
202  bdd = bdd.ExistAbstract(manager.computeCube(out));
203  /* limit the grid points in the grid */
204  for(const auto& interval : m_bdd_interval)
205  bdd = bdd & interval.get_all_elements();
206  /* init the vector of grid points to be returned */
207  abs_type no_gp = get_size(manager,bdd);
208  std::vector<double> gp(no_gp*m_dim);
209  for(abs_type i=0; i<no_gp; i++) {
210  for(int j=0; j<m_dim; j++)
211  gp[i*m_dim+j]=m_first[j];
212  }
213  /* set up iteration to iterate over BDD cubes */
214  DdManager* dd = manager.getManager();
215  int *cube;
216  CUDD_VALUE_TYPE value;
217  DdGen *gen;
218  abs_type counter=0;
219  /* iterate over BDD cubes */
220  Cudd_ForeachCube(dd,bdd.getNode(),gen,cube,value) {
221  abs_type offset=1;
222  for(int i=0; i<m_dim; i++) {
223  unsigned int no_vars = m_bdd_interval[i].get_no_bdd_vars();
224  for (unsigned int j=0; j<no_vars; j++) {
225  unsigned int id = m_bdd_interval[i].get_bdd_var_ids()[j];
226  if(cube[id]==1) {
227  for(abs_type k=0; k<offset; k++) {
228  gp[(counter+k)*m_dim+i]+=(abs_type{1}<<(no_vars-1-j))*m_eta[i];
229  }
230  }
231  /* take care of don't care */
232  if(cube[id]==2) {
233  for(abs_type k=0; k<offset; k++) {
234  for(int l=0; l<=i; l++) {
235  gp[(counter+k+offset)*m_dim+l]=gp[(counter+k)*m_dim+l];
236  }
237  }
238  for(abs_type k=0; k<offset; k++) {
239  gp[(counter+k+offset)*m_dim+i]+=(abs_type{1}<<(no_vars-1-j))*m_eta[i];
240  }
241  offset=(offset<<1);
242  }
243  }
244  }
245  counter+=offset;
246  }
247  /* reactivate reordering if it was enabled */
248  if(method!=nullptr)
249  manager.AutodynEnable(*method);
250  return gp;
251  }

309  codomain.push_back(i);

318  restricted = restricted.ExistAbstract(manager.computeCube(set_dom.get_bdd_vars()));
