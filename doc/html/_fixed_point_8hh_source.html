48  /* function: FixedPoint
49  *
50  * initialize the FixedPoint object with a <SymbolicModel> containing the
51  * transition relation
52  */
53  FixedPoint(SymbolicModel *symbolicModel) {
54  symbolicModel_=symbolicModel;
55  ddmgr_=symbolicModel_->ddmgr_;
56  /* the permutation array */
57  size_t n=ddmgr_->ReadSize();
58  permute_ = new int[n];
59  for(size_t i=0; i<n; i++)
60  permute_[i]=i;
61  for(size_t i=0; i<symbolicModel_->nssVars_; i++)
62  permute_[symbolicModel_->preVars_[i]]=symbolicModel_->postVars_[i];
63  /* create a cube with the input Vars */
64  BDD* vars = new BDD[symbolicModel_->nisVars_];
65  for (size_t i=0; i<symbolicModel_->nisVars_; i++)
66  vars[i]=ddmgr_->bddVar(symbolicModel_->inpVars_[i]);
67  cubeInput_ = ddmgr_->bddComputeCube(vars,NULL,symbolicModel_->nisVars_);
68  delete[] vars;
69  /* create a cube with the post Vars */
70  vars = new BDD[symbolicModel_->nssVars_];
71  for (size_t i=0; i<symbolicModel_->nssVars_; i++)
72  vars[i]=ddmgr_->bddVar(symbolicModel_->postVars_[i]);
73  cubePost_ = ddmgr_->bddComputeCube(vars,NULL,symbolicModel_->nssVars_);
74  delete[] vars;
75 
76  /* copy the transition relation */
77  R_=symbolicModel_->transitionRelation_;
78  RR_=R_.ExistAbstract(cubePost_);
79  }

84  /* function: pre
85  *
86  * computes the enforcable predecessor
87  *
88  */
89  BDD pre(BDD Z) {
90  /* project onto state alphabet */
91  Z=Z.ExistAbstract(cubePost_*cubeInput_);
92  /* swap variables */
93  Z=Z.Permute(permute_);
94  /* find the (state, inputs) pairs with a post outside the safe set */
95  BDD nZ = !Z;
96  BDD F = R_.AndAbstract(nZ,cubePost_);
97  /* the remaining (state, input) pairs make up the pre */
98  BDD nF = !F;
99  BDD preZ= RR_.AndAbstract(nF,cubePost_);
100  return preZ;
101  }
102  return Z;
103  }

111  /* check if target is a subset of the state space */
112  std::vector<unsigned int> sup = T.SupportIndices();
113  for(size_t i=0; i<sup.size();i++) {
114  int marker=0;
115  for(size_t j=0; j<symbolicModel_->nssVars_; j++) {
116  if (sup[i]==symbolicModel_->preVars_[j])
117  marker=1;
118  }
119  if(!marker) {
120  std::ostringstream os;
121  os << "Error: reach: the target set depends on variables outside of the state space.";
122  throw std::invalid_argument(os.str().c_str());
123  }
124  }
125  if(verbose)
126  std::cout << "Iterations: ";
127 
128  BDD Z = ddmgr_->bddOne();
129  BDD ZZ = ddmgr_->bddZero();
130  /* the controller */
131  BDD C = ddmgr_->bddZero();
132  /* as long as not converged */
133  size_t i;
134  for(i=1; ZZ != Z; i++ ) {
135  Z=ZZ;

139  /* add new (state/input) pairs to the controller */
140  C=C | N;

148  }
